How to save:

1. Open this file in Chrome: C:\InviteWala\invitewala-fresh\assets\logo-generator.html

2. Press F12 → Console tab → paste this code:

// Install html2canvas first by pasting:
var s = document.createElement('script');
s.src = 'https://cdnjs.cloudflare.com/ajax/libs/html2canvas/1.4.1/html2canvas.min.js';
document.head.appendChild(s);

// Then after it loads (wait 2 sec), run:
setTimeout(() => {
  const el = document.querySelector('.icon-wrap');
  el.style.transform = 'none';
  html2canvas(el, {width:1024,height:1024,scale:1}).then(c => {
    const a = document.createElement('a');
    a.download = 'icon.png';
    a.href = c.toDataURL('image/png');
    a.click();
    el.style.transform = '';
  });
}, 2000);
